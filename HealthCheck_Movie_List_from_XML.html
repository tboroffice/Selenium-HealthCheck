store <testdata> xmlData
store <testdata> ListData
loadTestData file://c:/Users/Strider/Documents/Work/Rovi/Mergeshows.xml
while !testdata.EOF()
nextTestData
store ${sourceID} source_ID_var
store ${title} Fan_title_var
storeEval storedVars.Fan_title_var.replace(/&/g,"&amp;").replace(/[>]/g,"&gt;").replace(/[<]/g,"&lt;").replace(/[`’]/g,"&apos;").replace(/'/g,"&apos;").replace(/"/g,"&quot;") Fan_title_var
store ${Fan_List} source_list
storeEval storedVars.source_list.replace(/&/g,"&amp;").replace(/[>]/g,"&gt;").replace(/[<]/g,"&lt;").replace(/[`’]/g,"&apos;").replace(/'/g,"&apos;").replace(/"/g,"&quot;") source_list
store ${Release_details} viewing_list
storeEval storedVars.viewing_list.replace(/&/g,"&amp;").replace(/[>]/g,"&gt;").replace(/[<]/g,"&lt;").replace(/[`’]/g,"&apos;").replace(/'/g,"&apos;").replace(/"/g,"&quot;") viewing_list
gotoIf storedVars.source_ID_var == "0" no_id_label
open http://api.fanhattan.com/3.1/content/movies/${sourceID}
storeText //div[@id='json']/ul/li[6]/span[2] runtime
storeText //div[@id='json']/ul/li[12]/span[2] release_date
storeEval storedVars.release_date.replace(/["]/g,"") release_date
storeText //div[@id='json']/ul/li[16]/span[2] content_rating
storeEval storedVars.content_rating.replace(/["]/g,"") content_rating
storeCssCount ul.obj.collapsible > li:contains("genres") > ul.array.collapsible > li genre_count
echo ${genre_count}
echo ${release_date}
label cast_and_crew_metadata_label
open http://api.fanhattan.com/3.1/content/movies/${sourceID}/cast
storeText //div[@id='json']/ul/li[3]/span[2] cast_count
open http://api.fanhattan.com/3.1/content/movies/${sourceID}/crew
storeText //div[@id='json']/ul/li[3]/span[2] crew_count
label external_ratings_label
open http://api.fanhattan.com/3.1/content/movies/${sourceID}/external_ratings
storeText //div[@id='json']/ul/li[3]/span[2] external_ratings_count
label image_check_label
open http://api.fanhattan.com/3.1/content/movies/${sourceID}/images
storeCssCount css=ul.obj.collapsible > li:contains("backdrops") > ul.array.collapsible > li image_backdrops
echo ${image_backdrops}
storeCssCount css=ul.obj.collapsible > li:contains("posters") > ul.array.collapsible > li image_posters
echo ${image_posters}
goto xml_write_label
label no_id_label
label xml_write_label
storeEval storedVars.xmlData+ '<test source="${source_list}" title="${Fan_title_var}" sourceID="${sourceID}" released="${release_date}" rated="${content_rating}" genres="${genre_count}" cast="${cast_count}" crew="${crew_count}" backdrops="${image_backdrops}" posters="${image_posters}" external_ratings="${external_ratings_count}" sources="${viewing_list}"/>' xmlData
endWhile
runScript javascript{storedVars['xmlData'] += '</testdata>';}
label exit_prompt_label
storeEval prompt("XML Data",storedVars.xmlData) output_string_var
echo ${xmlData}
